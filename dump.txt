student details - [roll num , image, class(foregion key to class master) , session(foregion key to acadmic year master) ,contact number , details , address ,guardian name , guardian number , Dob,]
CREATE TABLE students (
  student_id INT AUTO_INCREMENT PRIMARY KEY,
  roll_no VARCHAR(50) NOT NULL UNIQUE,
  student_name VARCHAR(100) NOT NULL,
  image VARCHAR(255) NULL,
  class_id INT NOT NULL,
  session_id INT NOT NULL,
  contact_number VARCHAR(15) NOT NULL,
  address TEXT,
  details TEXT,
  guardian_name VARCHAR(100),
  guardian_number VARCHAR(15),
  dob DATE,
  
  CONSTRAINT fk_class 
    FOREIGN KEY (class_id) REFERENCES class_master(class_id)
    ON UPDATE CASCADE ON DELETE RESTRICT,

  CONSTRAINT fk_session 
    FOREIGN KEY (session_id) REFERENCES session_master(session_id)
    ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB;
